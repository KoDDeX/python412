### 2. –ì–µ–Ω–µ—Ä–∞—Ç–æ—Ä –º–µ–º–∞


### 3. –ü–æ–∏—Å–∫ –ø–∞–ª–∏–Ω–¥—Ä–æ–º–æ–≤

```python
text = "–ê —Ä–æ–∑–∞ —É–ø–∞–ª–∞ –Ω–∞ –ª–∞–ø—É –ê–∑–æ—Ä–∞. –î–µ–¥ —á–∏—Ç–∞–µ—Ç. –®–∞–ª–∞—à —Å—Ç–æ–∏—Ç."

def clean_text(text):
    result = ""
    for char in text.lower():
        if char.isalpha():
            result += char
    return result

words = text.split()
palindromes = []
stats = {}  # –î–ª—è —Å—Ç–∞—Ç–∏—Å—Ç–∏–∫–∏ –ø–æ –¥–ª–∏–Ω–µ

for word in words:
    clean_word = clean_text(word)
    if len(clean_word) >= 3:
        if clean_word == clean_word[::-1]:
            palindromes.append(word)
            length = len(clean_word)
            if length in stats:
                stats[length] += 1
            else:
                stats[length] = 1

print("–ù–∞–π–¥–µ–Ω–Ω—ã–µ –ø–∞–ª–∏–Ω–¥—Ä–æ–º—ã:", palindromes)
print("–°—Ç–∞—Ç–∏—Å—Ç–∏–∫–∞ –ø–æ –¥–ª–∏–Ω–µ:", stats)
```

### 4. –¢–µ–∫—Å—Ç–æ–≤—ã–π –≥—Ä–∞–¥–∏–µ–Ω—Ç

```python

result = []

# –° –∫–∞–∂–¥—ã–º —Å–ª–æ–≤–æ–º —É–≤–µ–ª–∏—á–∏–≤–∞–µ–º –∫–æ–ª–∏—á–µ—Å—Ç–≤–æ –∑–∞–≥–ª–∞–≤–Ω—ã—Ö –±—É–∫–≤

    new_word = ""
    # –ö–æ–ª–∏—á–µ—Å—Ç–≤–æ –∑–∞–≥–ª–∞–≤–Ω—ã—Ö –±—É–∫–≤ –∑–∞–≤–∏—Å–∏—Ç –æ—Ç –ø–æ–∑–∏—Ü–∏–∏ —Å–ª–æ–≤–∞
    caps_count = int((i + 1) * len(word) / len(words))
    
    for j, char in enumerate(word):
        if j < caps_count:
            new_word += char.upper()
        else:
            new_word += char.lower()
    result.append(new_word)

# –î–æ–±–∞–≤–ª—è–µ–º —Å–∏–º–≤–æ–ª—ã –≥—Ä–∞–¥–∏–µ–Ω—Ç–∞

```

```python
    for word in words:

    # –î–æ–±–∞–≤–ª—è–µ–º —Ç–æ–ª—å–∫–æ –µ—Å–ª–∏ –Ω–∞—à–ª–∏ –º–∏–Ω–∏–º—É–º 2 —Ä–∏—Ñ–º—É—é—â–∏—Ö—Å—è —Å–ª–æ–≤–∞
    if len(current_rhyme) > 1 and current_rhyme not in rhymes:
        rhymes.append(current_rhyme)

# –í—ã–≤–æ–¥–∏–º —Ä–µ–∑—É–ª—å—Ç–∞—Ç
for rhyme_group in rhymes:
    ending = rhyme_group[0][-2:]
    print(f'–†–∏—Ñ–º–∞ "{ending}": {", ".join(rhyme_group)}')
```

```python
    spaces_count = line.count(" ")
    clean_line = " ".join(line.split())
    stats["removed_spaces"] += spaces_count - clean_line.count(" ")
    
    # –ü—Ä–æ–ø—É—Å–∫–∞–µ–º –∫–æ—Ä–æ—Ç–∫–∏–µ —Å—Ç—Ä–æ–∫–∏ –∏ —á–∏—Å–ª–æ–≤—ã–µ —Å—Ç—Ä–æ–∫–∏
    if len(clean_line) < 3:
        stats["removed_lines"] += 1
        continue
    if clean_line.isdigit():
        stats["removed_numbers"] += 1
        continue
        
    if clean_line:  # –ü—Ä–æ–ø—É—Å–∫–∞–µ–º –ø—É—Å—Ç—ã–µ —Å—Ç—Ä–æ–∫–∏
        filtered_lines.append(clean_line)

result = "\n".join(filtered_lines)
print(result)
print("–°—Ç–∞—Ç–∏—Å—Ç–∏–∫–∞:", stats)
```

### 7. –ì–µ–Ω–µ—Ä–∞—Ç–æ—Ä –Ω–∏–∫–Ω–µ–π–º–æ–≤

```python
# –ó–∞–º–µ–Ω–∞ –±—É–∫–≤ –Ω–∞ —Ü–∏—Ñ—Ä—ã
replacements = {
    "–æ": "0",
    "–∑": "3",
    "–µ": "3",
    "–∞": "4",
    "–∏": "1",
    "—Å": "5"
}

result = ""
upper = True

# –ü—Ä–µ–æ–±—Ä–∞–∑—É–µ–º —Ç–µ–∫—Å—Ç
for char in text.lower():
    if char in replacements:
        result += replacements[char]
    elif char.isalpha():
        if upper:
            result += char.upper()
        else:
            result += char
        upper = not upper
    else:
        result += char

# –î–æ–±–∞–≤–ª—è–µ–º —Å–ª—É—á–∞–π–Ω—ã–π —Å–∏–º–≤–æ–ª
import random
result = result.replace(" ", random.choice(symbols))

if len(result) > 15:
    result = result[:15]

print(result)
```
